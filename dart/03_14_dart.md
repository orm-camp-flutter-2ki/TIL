  void a();

abstract class Y implements X {
  void b();
}

  @override
  void a() {
    print('Aa');
  }

  @override
  void b() {
    print('Ab');
  }

  void c() {
    print('Ac');
  }
}

class B extends Y {
  @override
  void a() {
    print('Ba');
  }

  @override
  void b() {
    print('Bb');
  }

  void c() {
    print('Bc');
  }
}

  X obj = A();
  Y y1 = A();
  Y y2 = B();

  List<Y> list = [];
  list.add(y1);
  list.add(y2);

  for(Y i in list) {
    i.b();
  }
  y1.a();
  y2.a();
  obj.a();
}
